快读模板

int read()
{
    int a = 0, b = 1;
    char c = getchar();
    while (c >= '9' || c <= '0')
    {
        if (c == '-')
            b = -1;
        c = getchar();
    }
    while (c <= '9' && c >= '0')
    {
        a = a * 10 + c - '0';
        c = getchar();
    }
    return a * b;
}

快写模板

void printi(int x)
{
    if (x / 10)
        printi(x / 10);
    putchar(x % 10 + '0');
}
